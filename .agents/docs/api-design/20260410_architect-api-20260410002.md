### 1.2 Backup API - 备份数据操作

#### `backup:exportData`

将密码库数据导出到指定文件路径。

**参数**：
```typescript
{
  /** 用户选择的保存路径 */
  filePath: string
}
```

**返回值**：
```typescript
interface ExportResult {
  /** 是否成功 */
  success: boolean
  /** 错误信息 */
  error?: string
  /** 导出的条目数量 */
  entryCount: number
  /** 导出的文件路径 */
  filePath: string
}
```

**主进程实现逻辑**：
```typescript
ipcMain.handle('backup:exportData', async (_event, filePath: string): Promise<ExportResult> => {
  try {
    // 1. 读取数据库全部数据
    const entries = getEntries()
    const masterKeyVerify = getMasterKeyVerify()
    const appVersion = app.getVersion()

    // 2. 构建备份数据
    const backupData: BackupData = {
      version: 1,
      exportedAt: new Date().toISOString(),
      appVersion,
      masterKeyVerify,
      entries,
    }

    // 3. 写入文件
    const fs = await import('fs/promises')
    await fs.writeFile(filePath, JSON.stringify(backupData, null, 2), 'utf-8')

    return {
      success: true,
      entryCount: entries.length,
      filePath,
    }
  } catch (error: any) {
    return {
      success: false,
      error: error.message || '导出失败',
      entryCount: 0,
      filePath,
    }
  }
})
```

#### `backup:validateBackup`

验证备份文件的主密码是否匹配当前密码库。

**参数**：
```typescript
{
  /** 备份文件路径 */
  filePath: string
  /** 用户输入的主密码 */
  password: string
}
```

**返回值**：
```typescript
interface ValidateResult {
  /** 是否验证成功 */
  success: boolean
  /** 错误信息 */
  error?: string
  /** 是否密码匹配 */
  isValid: boolean
}
```

**主进程实现逻辑**：
```typescript
ipcMain.handle('backup:validateBackup', async (_event, filePath: string, password: string): Promise<ValidateResult> => {
  try {
    // 1. 读取并解析备份文件
    const fs = await import('fs/promises')
    const content = await fs.readFile(filePath, 'utf-8')
    const backupData: BackupData = JSON.parse(content)

    // 2. 格式验证
    if (!backupData.version || !backupData.masterKeyVerify) {
      return { success: true, isValid: false, error: '备份文件格式不正确' }
    }

    // 3. 使用备份文件的验证数据验证密码
    const isValid = verifyPassword(backupData.masterKeyVerify, password)

    return { success: true, isValid }
  } catch (error: any) {
    return {
      success: false,
      isValid: false,
      error: error.message || '验证失败',
    }
  }
})
```

**安全说明**：此接口将用户输入的密码通过IPC传到主进程进行验证。这与现有的 `crypto:verifyPassword` 通道行为一致，密码仅存在于内存中，不被持久化。

```typescript
ipcMain.handle('backup:parseBackup', async (_event, filePath: string): Promise<ParseResult> => {
  try {
    const fs = await import('fs/promises')
    const content = await fs.readFile(filePath, 'utf-8')
    const backupData: BackupData = JSON.parse(content)

    // 格式基础验证
    if (typeof backupData.version !== 'number') {
      return { success: false, error: '无效的备份文件格式' }

    return {
      success: true,
      version: backupData.version,
      exportedAt: backupData.exportedAt,
      appVersion: backupData.appVersion,
      entryCount: backupData.entries?.length ?? 0,
    }
  } catch (error: any) {
    return {
      success: false,
      error: error.message || '文件解析失败',
    }
  }
})
```

#### `backup:checkConflicts`

检查备份文件中的条目与本地数据库条目的ID冲突。

**参数**：
```typescript
{
  /** 备份文件路径 */
  filePath: string
}
```

**返回值**：
```typescript
interface ConflictCheckResult {
  /** 是否成功 */
  success: boolean
  /** 错误信息 */
  error?: string
  /** 冲突条目列表（仅含元信息，不含密码） */
  conflicts: ConflictEntry[]
  /** 无冲突的备份条目数量 */
  noConflictCount: number
}

interface ConflictEntry {
  /** 条目ID */
  id: string
  /** 本地条目信息（不含密码） */
  local: EntrySummary
  /** 备份条目信息（不含密码） */
  backup: EntrySummary
}

interface EntrySummary {
  title: string
  username: string
  createdAt: number
  updatedAt: number
}
```

**主进程实现逻辑**：
```typescript
ipcMain.handle('backup:checkConflicts', async (_event, filePath: string): Promise<ConflictCheckResult> => {
  try {
    // 1. 读取并解析备份文件
    const fs = await import('fs/promises')
    const content = await fs.readFile(filePath, 'utf-8')
    const backupData: BackupData = JSON.parse(content)

    // 2. 获取本地条目
    const localEntries = getEntries()
    const localIdMap = new Map(localEntries.map(e => [e.id, e]))

    // 3. 检测冲突
    const conflicts: ConflictEntry[] = []
    let noConflictCount = 0

    for (const backupEntry of backupData.entries) {
      const localEntry = localIdMap.get(backupEntry.id)
      if (localEntry) {
        // 冲突：仅提取元信息
        conflicts.push({
          id: backupEntry.id,
          local: {
            title: localEntry.title,
            username: localEntry.username,
            createdAt: localEntry.createdAt,
            updatedAt: localEntry.updatedAt,
          },
          backup: {
            title: backupEntry.title,
            username: backupEntry.username,
            createdAt: backupEntry.createdAt,
            updatedAt: backupEntry.updatedAt,
          },
        })
      } else {
        noConflictCount++
      }
    }

    return { success: true, conflicts, noConflictCount }
  } catch (error: any) {
    return {
      success: false,
      error: error.message || '冲突检测失败',
      conflicts: [],
      noConflictCount: 0,
    }
  }
})
```

#### `backup:importData`

执行导入操作，根据冲突解决方案写入数据。

**参数**：
```typescript
{
  /** 备份文件路径 */
  filePath: string
  /** 冲突解决方案列表 */
  resolutions: ConflictResolution[]
}

interface ConflictResolution {
  /** 条目ID */
  entryId: string
  /** 处理方式 */
  action: 'override' | 'skip' | 'keepboth'
}
```

**返回值**：
```typescript
interface ImportResult {
  /** 是否成功 */
  success: boolean
  /** 错误信息 */
  error?: string
  /** 导入统计 */
  stats: {
    /** 新增条目数（无冲突条目 + keepboth条目） */
    added: number
    /** 覆盖条目数 */
    overridden: number
    /** 跳过条目数 */
    skipped: number
    /** 保留两者条目数 */
    keptBoth: number
  }
}
```

**主进程实现逻辑**：
```typescript
ipcMain.handle('backup:importData', async (_event, filePath: string, resolutions: ConflictResolution[]): Promise<ImportResult> => {
  // 1. 内存备份当前数据（用于回滚）
  const backup = {
    entries: JSON.parse(JSON.stringify(getEntries())),
    masterKeyVerify: getMasterKeyVerify(),
  }

  try {
    // 2. 读取备份文件
    const fs = await import('fs/promises')
    const content = await fs.readFile(filePath, 'utf-8')
    const backupData: BackupData = JSON.parse(content)

    // 3. 获取本地条目
    const localEntries = getEntries()
    const localIdSet = new Set(localEntries.map(e => e.id))

    // 4. 构建解决方案映射
    const resolutionMap = new Map(resolutions.map(r => [r.entryId, r.action]))

    // 5. 统计计数
    let added = 0
    let overridden = 0
    let skipped = 0
    let keptBoth = 0

    // 6. 处理每条备份条目
    for (const entry of backupData.entries) {
      if (localIdSet.has(entry.id)) {
        // 冲突条目
        const action = resolutionMap.get(entry.id) ?? 'skip' // 默认跳过

        if (action === 'override') {
          await updateEntry(entry.id, entry)
          overridden++
        } else if (action === 'keepboth') {
          const newEntry = { ...entry, id: crypto.randomUUID() }
          await addEntry(newEntry)
          keptBoth++
        } else {
          skipped++
        }
      } else {
        // 无冲突条目 - 直接添加
        await addEntry(entry)
        added++
      }
    }

    return {
      success: true,
      stats: { added, overridden, skipped, keptBoth },
    }
  } catch (error: any) {
    // 7. 回滚
    try {
      const db = getDatabase()
      db.data.entries = backup.entries
      db.data.masterKeyVerify = backup.masterKeyVerify!
      await db.write()
    } catch (rollbackError) {
      console.error('Import rollback failed:', rollbackError)
    }

    return {
      success: false,
      error: error.message || '导入失败，数据已恢复',
      stats: { added: 0, overridden: 0, skipped: 0, keptBoth: 0 },
    }
  } finally {
    // 8. 清除内存备份
    backup.entries = []
    backup.masterKeyVerify = null
  }
})
```

## 5. 默认文件名生成

```typescript
/**
 * 生成默认导出文件名
 * 格式：PassLock_Backup_YYYYMMDD_HHMMSS.json
 */
function generateBackupFileName(): string {
  const now = new Date()
  const pad = (n: number) => n.toString().padStart(2, '0')
  const dateStr = `${now.getFullYear()}${pad(now.getMonth() + 1)}${pad(now.getDate())}`
  const timeStr = `${pad(now.getHours())}${pad(now.getMinutes())}${pad(now.getSeconds())}`
  return `PassLock_Backup_${dateStr}_${timeStr}.json`
}
```

## 6. 错误码规范
